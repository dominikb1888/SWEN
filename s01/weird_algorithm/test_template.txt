#[test]
fn test_{test_name}() {
    let input = r#"{input_content}"#;
    let expected_output = r#"{expected_output}"#;

    let mut cmd = Command::cargo_bin("{bin_name}").unwrap();

    let assert_obj = cmd
        .write_stdin(input)
        .assert()
        .success();

    let actual_stdout = String::from_utf8_lossy(&assert_obj.get_output().stdout);

    assert_eq!(
        actual_stdout.trim(),
        expected_output.trim(),
        "Test failed for \"{}\".\nExpected:\n{}\nGot:\n{}",
        "{input_file}",
        expected_output.trim(),
        actual_stdout.trim()
    );
}
